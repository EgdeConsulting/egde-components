import { MultiSelect } from 'egde-components';
import { useState } from 'react';

export const Template = (args) => {
    const [value, setValue] = useState([]);
    const handleChange = (e) => {
        setValue(e);
    };
    return (
        <div>
            <MultiSelect {...args} value={value} onChange={handleChange} />
        </div>
    );
};
