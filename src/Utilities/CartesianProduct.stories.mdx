```ts
export function createCartesianProduct(...arrayList): string[] {
    if (arrayList.length > 1) {
        return arrayList.reduce((a, b) =>
            a.flatMap((d) => b.map((e) => [d, e].flat())),
        );
    } else {
        let newArrayList = [];

        arrayList[0].forEach((item) => {
            newArrayList.push([item]);
        });

        return newArrayList;
    }
}
```
