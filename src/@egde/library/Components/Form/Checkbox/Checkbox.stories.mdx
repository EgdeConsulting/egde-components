import { Meta, Story, Canvas } from '@storybook/addon-docs';
import { LibPath, DocsPageHeader, DocsPageBodyComponents } from '@egde/library';
import { Checkbox } from '@egde/components';
import { useState } from 'react';

export const Template = (args) => {
    const [values, setValues] = useState([...args.value]);
    const handleOnChange = (newValue) => {
        setValues(newValue);
    };
    return (
        <div>
            <Checkbox {...args} onChange={handleOnChange} value={values} />
        </div>
    );
};

export const componentName = 'Checkbox';

<Meta
    title={LibPath.Form + componentName}
    component={Checkbox}
    args={{
        direction: 'column',
        options: ['Option One', 'Option Two', 'Option Three'],
        value: [0],
        onChange: () => {},
        invalidText: '',
        isDisabled: false,
        backgroundColor: '#ffffff',
        autoFocus: false,
        label: 'Checkbox',
        labelTextMargin: '10px',
        captionText: '',
        captionTextMargin: '0px',
        margin: '0px',
        padding: '0px',
        width: '100%',
        minWidth: '0px',
        maxWidth: '100%',
    }}
/>

<DocsPageHeader
    title={componentName}
    description="This component is used to create a checklist where multiple answers can be checked off."
    addImportantNote={false}
    variant="stable"
/>

</Canvas>
