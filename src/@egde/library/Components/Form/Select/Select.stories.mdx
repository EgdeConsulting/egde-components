import { Meta, Story, Canvas } from '@storybook/addon-docs';
import { LibPath, DocsPageHeader, DocsPageBodyComponents } from '@egde/library';
import { Select } from '@egde/components';
import { useState } from 'react';

export const Template = (args) => {
    const [value, setValue] = useState();
    const handleChange = (e) => {
        setValue(e);
    };
    return (
        <div>
            <Select {...args} value={value} onChange={handleChange} />
        </div>
    );
};

export const componentName = 'Select';

<Meta
    title={LibPath.Form + 'Select'}
    component={Select}
    args={{
        label: 'Title',
        options: [
            { label: 'Option 1', value: 0 },
            { label: 'Option 2', value: 1 },
            { label: 'Option 3', value: 2 },
        ],
        value: {
            label: '',
            value: 0,
        },
        placeholder: 'Select an option:',
        filterText: '',
        isDisabled: false,
        labelTextMargin: '0',
        invalidText: '',
        captionText: 'Caption text.',
        captionTextMargin: '0',
        margin: '10px',
        padding: '0',
        width: '100%',
        minWidth: '0',
        maxWidth: '100%',
        onChange: () => {},
    }}
/>

<DocsPageHeader
    title={componentName}
    description="This component is used to create a vertical dropdown-menu where one option can be selected be clicking it."
    addImportantNote={false}
    variant="stable"
/>

<Canvas withToolbar>
    <Story name="Base" height="250px">
        {Template.bind({})}
    </Story>
</Canvas>
